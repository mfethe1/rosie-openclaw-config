## 4. Concrete Handoff Specs for Mack

### agent-memory.db Phase 1 Schema (ready to implement)

```sql
-- core memory table
CREATE TABLE agent_memories (
  id          INTEGER PRIMARY KEY AUTOINCREMENT,
  agent       TEXT NOT NULL,              -- rosie | mack | winnie | lenny
  cycle       TEXT NOT NULL,              -- e.g. 2026-02-17-19-winnie
  topic       TEXT NOT NULL,              -- deployment | handoff | blocker | decision | ...
  body        TEXT NOT NULL,              -- full memory text
  source_file TEXT,                       -- e.g. outputs/2026-02-17-19-winnie.md
  tags        TEXT,                       -- comma-separated tags
  created_at  TEXT NOT NULL DEFAULT (datetime('now'))
);

-- FTS5 index for body + tags
CREATE VIRTUAL TABLE agent_memories_fts USING fts5(
  body,
  tags,
  content='agent_memories',
  content_rowid='id'
);

-- Keep FTS in sync
CREATE TRIGGER fts_insert AFTER INSERT ON agent_memories BEGIN
  INSERT INTO agent_memories_fts(rowid, body, tags) VALUES (new.id, new.body, new.tags);
END;

CREATE TRIGGER fts_delete AFTER DELETE ON agent_memories BEGIN
  INSERT INTO agent_memories_fts(agent_memories_fts, rowid, body, tags) VALUES ('delete', old.id, old.body, old.tags);
END;

CREATE TRIGGER fts_update AFTER UPDATE ON agent_memories BEGIN
  INSERT INTO agent_memories_fts(agent_memories_fts, rowid, body, tags) VALUES ('delete', old.id, old.body, old.tags);
  INSERT INTO agent_memories_fts(rowid, body, tags) VALUES (new.id, new.body, new.tags);
END;
```
